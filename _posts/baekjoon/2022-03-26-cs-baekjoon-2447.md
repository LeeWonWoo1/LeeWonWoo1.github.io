## 3. Python 문제풀이

```python
def get_stars(n) :
  arr = []
  for i in range(len(n) * 3) :
    if i // len(n) == 1 :
      arr.append(n[i % len(n)] + " " * len(n) + n[i % len(n)])
    else :
      arr.append(n[i % len(n)] * 3)
  return arr

resultStar = ["***", "* *", "***"]

n = int(input())
count = 0

while n != 3 :
  n //= 3
  count += 1

for _ in range(count) :
  resultStar = get_stars(resultStar)
for j in resultStar :
  print(j)
```
